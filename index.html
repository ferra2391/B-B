const SHEET_MENU = "Menu"; const SHEET_CAMERE = "Camere"; const SHEET_ORDINI = "Ordini"; const EMAIL_DESTINATARIO = "ferra2391@gmail.com"; const NOME_BB = "LG Estate Colazioni"; const MINUTI_MODIFICA = 10; // --- IMPOSTAZIONI ORARI --- const ORA_CHIUSURA = 20; // Chiude alle 20:00 const ORA_RIAPERTURA = 10; // Riapre alle 10:00 del mattino successivo function isOrarioValido_() { const h = new Date().getHours(); // Se l'ora è maggiore o uguale a 20, oppure minore di 10, è chiuso. return !(h >= ORA_CHIUSURA || h < ORA_RIAPERTURA); } function doGet(e) { try { const action = String(e.parameter.action || "").trim(); const ss = SpreadsheetApp.getActive(); if (action === "init") { const struttura = String(e.parameter.struttura || "").trim(); const camera = String(e.parameter.camera || "").trim(); const key = String(e.parameter.key || "").trim(); const orarioValido = isOrarioValido_(); const menuData = ss.getSheetByName(SHEET_MENU).getDataRange().getValues(); const camereData = ss.getSheetByName(SHEET_CAMERE).getDataRange().getValues(); const ordiniSheet = ss.getSheetByName(SHEET_ORDINI); const ordiniData = ordiniSheet ? ordiniSheet.getDataRange().getValues() : []; // LETTURA NUOVO MENU (Tipologia, Prodotto, SubProdotto, Emoji, Attivo) const menuFlat = menuData.slice(1) .filter(r => String(r[4] || "").toUpperCase().trim() === "SI") .map(r => ({ tipologia: String(r[0] || "").trim(), prodotto: String(r[1] || "").trim(), sub: String(r[2] || "").trim(), emoji: String(r[3] || "🍽️").trim() })); // Raggruppamento per Frontend const menuStrutturato = []; const tipiMap = {}; menuFlat.forEach(item => { if (!tipiMap[item.tipologia]) { tipiMap[item.tipologia] = { tipologia: item.tipologia, prodotti: [] }; menuStrutturato.push(tipiMap[item.tipologia]); } const prodList = tipiMap[item.tipologia].prodotti; let prodObj = prodList.find(p => p.nome === item.prodotto); if (!prodObj) { prodObj = { nome: item.prodotto, emoji: item.emoji, varianti: [] }; prodList.push(prodObj); } if (item.sub) prodObj.varianti.push(item.sub); }); const isValida = camereData.slice(1).some(r => String(r[0]).trim() === struttura && String(r[1]).trim() === camera && String(r[2]).trim() === key && String(r[3]).toUpperCase().trim() === "SI" ); if (!isValida) return json_({ ok: false, error: "QR non autorizzato", menu: menuStrutturato }); let ordineAperto = null; const now = new Date(); for (let i = ordiniData.length - 1; i >= 1; i--) { const row = ordiniData[i]; const scadenza = new Date(row[1]); if (String(row[2]) === struttura && String(row[3]) === camera && now <= scadenza) { ordineAperto = { orderId: String(row[4]), prodotti: String(row[5]).split(",").map(x => x.trim()).filter(Boolean), note: String(row[6]), stato: String(row[7]), secondiRimanenti: Math.max(0, Math.floor((scadenza - now) / 1000)) }; break; } } return json_({ ok: true, menu: menuStrutturato, ordine: ordineAperto, fuoriOrario: !orarioValido }); } return json_({ ok: false, error: "Azione GET non valida." }); } catch (err) { return json_({ ok: false, error: err.message }); } } function doPost(e) { try { const data = JSON.parse(e.postData.contents || "{}"); const action = String(data.action || "").trim(); if (action === "creaOrdine") return json_(creaOrdine_(data)); if (action === "modificaOrdine") return json_(modificaOrdine_(data)); return json_({ ok: false, error: "Azione POST non valida." }); } catch (err) { return json_({ ok: false, error: err.message }); } } function creaOrdine_(data) { if (!isOrarioValido_()) return { ok: false, error: `Servizio chiuso. Ricorda di ordinare entro le ${ORA_CHIUSURA}:00.` }; const struttura = String(data.struttura || "").trim(); const camera = String(data.camera || "").trim(); const key = String(data.key || "").trim(); const prodotti = Array.isArray(data.prodotti) ? data.prodotti : []; const note = limitaTesto_(String(data.note || "").trim(), 500); const ospiti = parseInt(data.ospiti) || 1; // Riceve gli ospiti const maxScelte = ospiti * 2; // Calcola il limite if (!isCameraValida_(struttura, camera, key)) return { ok: false, error: "QR non autorizzato." }; if (!prodotti.length) return { ok: false, error: "Seleziona almeno un prodotto." }; if (prodotti.length > maxScelte) return { ok: false, error: `Puoi selezionare un massimo di ${maxScelte} pietanze.` }; const prodottiValidi = filtraProdottiValidi_(prodotti); if (!prodottiValidi.length) return { ok: false, error: "I prodotti selezionati non sono validi." }; const ordineAperto = getOrdineAperto_(struttura, camera); if (ordineAperto.ok && ordineAperto.ordine) return { ok: false, error: "Hai già un ordine modificabile.", ordine: ordineAperto.ordine }; const ordineOggi = getOrdineDiOggi_(struttura, camera); if (ordineOggi) return { ok: false, error: "Ordine già inviato oggi. Contatta la reception." }; const now = new Date(); const scadenza = new Date(now.getTime() + MINUTI_MODIFICA * 60 * 1000); const orderId = generaOrderId_(struttura, camera); const sh = SpreadsheetApp.getActive().getSheetByName(SHEET_ORDINI); if (!sh) return { ok: false, error: "Tab Ordini non trovata." }; sh.appendRow([now, scadenza, struttura, camera, orderId, prodottiValidi.join(", "), note, "INVIATO", now]); inviaMailOrdine_("NUOVO ORDINE", struttura, camera, orderId, prodottiValidi, note, now, scadenza, ospiti); return { ok: true, message: "Ordine inviato correttamente.", ordine: { struttura, camera, orderId, prodotti: prodottiValidi, note, stato: "INVIATO", scadenzaModifica: scadenza.toISOString(), secondiRimanenti: Math.max(0, Math.floor((scadenza.getTime() - new Date().getTime()) / 1000)) } }; } function modificaOrdine_(data) { if (!isOrarioValido_()) return { ok: false, error: `Servizio chiuso. Ordini consentiti fino alle ${ORA_CHIUSURA}:00.` }; const struttura = String(data.struttura || "").trim(); const camera = String(data.camera || "").trim(); const key = String(data.key || "").trim(); const orderId = String(data.orderId || "").trim(); const prodotti = Array.isArray(data.prodotti) ? data.prodotti : []; const note = limitaTesto_(String(data.note || "").trim(), 500); const ospiti = parseInt(data.ospiti) || 1; const maxScelte = ospiti * 2; if (!isCameraValida_(struttura, camera, key)) return { ok: false, error: "QR non autorizzato." }; if (!orderId) return { ok: false, error: "OrderId mancante." }; if (!prodotti.length) return { ok: false, error: "Seleziona almeno un prodotto." }; if (prodotti.length > maxScelte) return { ok: false, error: `Puoi selezionare un massimo di ${maxScelte} pietanze.` }; const prodottiValidi = filtraProdottiValidi_(prodotti); if (!prodottiValidi.length) return { ok: false, error: "Prodotti non validi." }; const sh = SpreadsheetApp.getActive().getSheetByName(SHEET_ORDINI); if (!sh) return { ok: false, error: "Tab Ordini non trovata." }; const values = sh.getDataRange().getValues(); const now = new Date(); for (let i = values.length - 1; i >= 1; i--) { const row = values[i]; if (String(row[2]).trim() === struttura && String(row[3]).trim() === camera && String(row[4]).trim() === orderId) { const scadenza = new Date(row[1]); if (now.getTime() > scadenza.getTime()) return { ok: false, error: "Tempo di modifica scaduto." }; sh.getRange(i + 1, 6).setValue(prodottiValidi.join(", ")); sh.getRange(i + 1, 7).setValue(note); sh.getRange(i + 1, 8).setValue("MODIFICATO"); sh.getRange(i + 1, 9).setValue(now); inviaMailOrdine_("MODIFICA ORDINE", struttura, camera, orderId, prodottiValidi, note, now, scadenza, ospiti); return { ok: true, message: "Ordine modificato.", ordine: { struttura, camera, orderId, prodotti: prodottiValidi, note, stato: "MODIFICATO", scadenzaModifica: scadenza.toISOString(), secondiRimanenti: Math.max(0, Math.floor((scadenza.getTime() - now.getTime()) / 1000)) } }; } } return { ok: false, error: "Ordine non trovato." }; } function isCameraValida_(struttura, camera, key) { const sh = SpreadsheetApp.getActive().getSheetByName(SHEET_CAMERE); if (!sh) return false; const values = sh.getDataRange().getValues().slice(1); return values.some(r => { return String(r[0] || "").trim() === struttura && String(r[1] || "").trim() === camera && String(r[2] || "").trim() === key && String(r[3] || "").toUpperCase().trim() === "SI"; }); } function getOrdineAperto_(struttura, camera) { const sh = SpreadsheetApp.getActive().getSheetByName(SHEET_ORDINI); if (!sh) return { ok: false, error: "Tab non trovata." }; const values = sh.getDataRange().getValues(); if (values.length < 2) return { ok: true, ordine: null }; const now = new Date(); for (let i = values.length - 1; i >= 1; i--) { const row = values[i]; const scadenza = new Date(row[1]); if (String(row[2]).trim() === struttura && String(row[3]).trim() === camera && now.getTime() <= scadenza.getTime()) { return { ok: true, ordine: { struttura: row[2], camera: row[3], orderId: String(row[4]).trim(), prodotti: String(row[5] || "").split(",").map(x => x.trim()).filter(Boolean), note: String(row[6] || "").trim(), stato: String(row[7] || "").trim(), scadenzaModifica: scadenza.toISOString(), secondiRimanenti: Math.max(0, Math.floor((scadenza.getTime() - now.getTime()) / 1000)) } }; } } return { ok: true, ordine: null }; } function getOrdineDiOggi_(struttura, camera) { const sh = SpreadsheetApp.getActive().getSheetByName(SHEET_ORDINI); if (!sh) return null; const values = sh.getDataRange().getValues(); const tz = Session.getScriptTimeZone(); const oggi = Utilities.formatDate(new Date(), tz, "yyyy-MM-dd"); for (let i = values.length - 1; i >= 1; i--) { if (!(values[i][0] instanceof Date)) continue; if (String(values[i][2]).trim() === struttura && String(values[i][3]).trim() === camera && Utilities.formatDate(values[i][0], tz, "yyyy-MM-dd") === oggi) return values[i]; } return null; } function filtraProdottiValidi_(prodotti) { const ss = SpreadsheetApp.getActive(); const menuData = ss.getSheetByName(SHEET_MENU).getDataRange().getValues().slice(1); const combinazioniValide = menuData .filter(r => String(r[4] || "").toUpperCase().trim() === "SI") .map(r => { const p = String(r[1] || "").trim(); const sub = String(r[2] || "").trim(); return sub ? `${p} - ${sub}` : p; }); return prodotti.map(x => String(x || "").trim()).filter(x => combinazioniValide.includes(x)); } function limitaTesto_(txt, max) { return txt ? txt.substring(0, max) : ""; } function generaOrderId_(struttura, camera) { const timestamp = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyyMMdd-HHmmss"); const rnd = Math.random().toString(36).substring(2, 7).toUpperCase(); return `${struttura.replace(/[^A-Z0-9]/gi, "").toUpperCase()}-${camera.replace(/[^A-Z0-9]/gi, "").toUpperCase()}-${timestamp}-${rnd}`; } function inviaMailOrdine_(tipo, struttura, camera, orderId, prodotti, note, data, scadenza, ospiti) { const subject = `${tipo} colazione - ${struttura} - Camera ${camera}`; const tz = Session.getScriptTimeZone(); const body = `B&B: ${NOME_BB}\nTipo: ${tipo}\nStruttura: ${struttura}\nCamera: ${camera}\nOspiti in camera: ${ospiti}\nOrder ID: ${orderId}\n\nProdotti Scelti:\n- ${prodotti.join("\n- ")}\n\nNote:\n${note || "Nessuna nota"}\n\nData: ${Utilities.formatDate(data, tz, "dd/MM/yyyy HH:mm:ss")}\nModificabile fino a: ${Utilities.formatDate(scadenza, tz, "dd/MM/yyyy HH:mm:ss")}`; MailApp.sendEmail({ to: EMAIL_DESTINATARIO, subject, body }); } function json_(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); }
